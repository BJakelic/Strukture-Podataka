int procitajDatoteku(char *imeDatoteke)                                     //1. funkcija
{
    int br=0;
    FILE *fp=NULL;
    char buffer[MAX_LINE]={0};
   
    fp=fopen(imeDatoteke,"r");
    if(!fp)
    {
        printf("Neuspjesno otvaranje datoteke.\n");
        return -1;
    }
   
    while(!feof(fp))
    {
        fgets(buffer,MAX_LINE,fp);
        br++;
    }
    fclose(fp);
    return br;
}
